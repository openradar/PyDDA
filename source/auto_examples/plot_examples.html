# Start the wind retrieval. This example only uses the mass continuity
# and data weighting constraints.

    Cm=256.0,
    Cx=0.0,
    Cy=0.0,
    Cz=0.0,
    Cb=0.0,
    frz=5000.0,

    upper_bc=1,

    engine="scipy",
)
